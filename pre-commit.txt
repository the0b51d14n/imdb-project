#!/usr/bin/env bash
# ══════════════════════════════════════════════════════════════════════════════
#  .git/hooks/pre-commit — Supinfo.TV
#  Bloque tout commit contenant des secrets ou fichiers sensibles.
#
#  Installation : cp pre-commit .git/hooks/pre-commit && chmod +x .git/hooks/pre-commit
# ══════════════════════════════════════════════════════════════════════════════

set -euo pipefail

RED='\033[0;31m'
YELLOW='\033[1;33m'
GREEN='\033[0;32m'
NC='\033[0m'

ERRORS=0

echo ""
echo "🔒 Supinfo.TV — Vérification de sécurité pre-commit..."
echo "────────────────────────────────────────────────────────"

# ── 1. Fichiers interdits ─────────────────────────────────────────────────────
FORBIDDEN_FILES=(
    ".env"
    "*.pem"
    "*.key"
    "*.p12"
    "*.pfx"
    "id_rsa"
    "id_ed25519"
    "*.sql.gz"
    "*.dump"
)

for pattern in "${FORBIDDEN_FILES[@]}"; do
    # Cherche dans les fichiers stagés
    matches=$(git diff --cached --name-only | grep -E "${pattern//\*/.*}" 2>/dev/null || true)
    if [ -n "$matches" ]; then
        echo -e "${RED}❌ BLOQUÉ — Fichier sensible détecté : ${matches}${NC}"
        ERRORS=$((ERRORS + 1))
    fi
done

# ── 2. Patterns de secrets dans le code ──────────────────────────────────────
SECRET_PATTERNS=(
    "TMDB_API_KEY\s*=\s*['\"][a-f0-9]{32}['\"]"
    "DB_PASS\s*=\s*['\"][^'\"]{4,}['\"]"
    "APP_SECRET\s*=\s*['\"][^'\"]{8,}['\"]"
    "PRICE_HMAC_SECRET\s*=\s*['\"][^'\"]{8,}['\"]"
    "password\s*=\s*['\"][^'\"]{4,}['\"]"
    "api_key\s*=\s*['\"][a-zA-Z0-9_\-]{20,}['\"]"
)

for pattern in "${SECRET_PATTERNS[@]}"; do
    matches=$(git diff --cached -U0 | grep '^\+' | grep -iE "$pattern" | grep -v "changeme\|your_\|example\|placeholder\|\\.env\\.example" || true)
    if [ -n "$matches" ]; then
        echo -e "${RED}❌ BLOQUÉ — Secret potentiel détecté :${NC}"
        echo "$matches" | head -5
        ERRORS=$((ERRORS + 1))
    fi
done

# ── 3. Fichier .env commité directement ──────────────────────────────────────
if git diff --cached --name-only | grep -qE '^\.env$'; then
    echo -e "${RED}❌ BLOQUÉ — Tentative de commit du fichier .env${NC}"
    ERRORS=$((ERRORS + 1))
fi

# ── 4. Vérification syntaxe PHP ──────────────────────────────────────────────
PHP_FILES=$(git diff --cached --name-only | grep '\.php$' || true)
if [ -n "$PHP_FILES" ]; then
    PHP_ERRORS=0
    while IFS= read -r file; do
        if [ -f "$file" ]; then
            if ! php -l "$file" > /dev/null 2>&1; then
                echo -e "${RED}❌ Erreur de syntaxe PHP : ${file}${NC}"
                php -l "$file" 2>&1 | head -3
                PHP_ERRORS=$((PHP_ERRORS + 1))
            fi
        fi
    done <<< "$PHP_FILES"
    if [ $PHP_ERRORS -gt 0 ]; then
        ERRORS=$((ERRORS + PHP_ERRORS))
    fi
fi

# ── 5. Vérification var_dump / die() debug ───────────────────────────────────
DEBUG_PATTERNS="var_dump\|print_r\|die(\|dd(\|dump(\|console\.log("
if git diff --cached -U0 | grep '^\+' | grep -v '^\+\+\+' | grep -qE "$DEBUG_PATTERNS"; then
    echo -e "${YELLOW}⚠️  Avertissement — Appels de debug détectés (var_dump, console.log…)${NC}"
    git diff --cached -U0 | grep '^\+' | grep -v '^\+\+\+' | grep -E "$DEBUG_PATTERNS" | head -5
    # Warning uniquement, pas bloquant
fi

# ── 6. Fichiers trop volumineux (>1MB) ───────────────────────────────────────
while IFS= read -r file; do
    if [ -f "$file" ]; then
        size=$(stat -c%s "$file" 2>/dev/null || stat -f%z "$file" 2>/dev/null || echo 0)
        if [ "$size" -gt 1048576 ]; then
            echo -e "${RED}❌ BLOQUÉ — Fichier trop volumineux (>1MB) : ${file} ($(( size / 1024 ))KB)${NC}"
            ERRORS=$((ERRORS + 1))
        fi
    fi
done < <(git diff --cached --name-only)

# ── Résultat ──────────────────────────────────────────────────────────────────
echo "────────────────────────────────────────────────────────"
if [ $ERRORS -gt 0 ]; then
    echo -e "${RED}🚫 Commit BLOQUÉ — ${ERRORS} erreur(s) de sécurité détectée(s).${NC}"
    echo -e "${YELLOW}   Corrigez les problèmes ci-dessus avant de commiter.${NC}"
    echo ""
    exit 1
fi

echo -e "${GREEN}✅ Vérifications de sécurité passées — Commit autorisé.${NC}"
echo ""
exit 0
